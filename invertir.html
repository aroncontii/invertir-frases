<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Invertir Frase</title>

    <script>

        function invertirFrase(frase) {

            // Invierte la frase y la devuelve

            return frase.split('').reverse().join('');

        }


        // Escucha mensajes del WebViewer

        window.addEventListener('message', function(event) {

            if (event.data && event.data.type === 'invertir') {

                const resultado = invertirFrase(event.data.text);

                // Devuelve el resultado a MIT App Inventor

                window.parent.postMessage({ type: 'resultado', text: resultado }, '*');

            }

        });

    </script>

</head>
